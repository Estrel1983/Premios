CREATE TABLE IF NOT EXISTS individuals (id INTEGER NOT NULL PRIMARY KEY, name TEXT NOT NULL, birthDate TEXT);
CREATE TABLE IF NOT EXISTS position (id INTEGER NOT NULL PRIMARY KEY, positionName TEXT NOT NULL);
CREATE TABLE IF NOT EXISTS employees (id INTEGER NOT NULL PRIMARY KEY, name TEXT NOT NULL, positionName TEXT NOT NULL, startDate TEXT NOT NULL, endDate TEXT, FOREIGN KEY (name) REFERENCES individuals(name), FOREIGN KEY (positionName) REFERENCES position(positionName));
CREATE TABLE IF NOT EXISTS premiosIncomLowLimit (id INTEGER NOT NULL PRIMARY KEY, positionName TEXT NOT NULL, limitSum INTEGER , FOREIGN KEY (positionName) REFERENCES position(positionName));
CREATE TABLE IF NOT EXISTS premiosPercent (id INTEGER NOT NULL PRIMARY KEY, positionName TEXT NOT NULL, percent REAL NULL, FOREIGN KEY (positionName) REFERENCES position(positionName));
CREATE TABLE IF NOT EXISTS revenuePerDay (id INTEGER NOT NULL PRIMARY KEY, date TEXT NOT NULL UNIQUE, revenueCash REAL, revenueNonCash REAL);
CREATE TABLE IF NOT EXISTS employeesPerDay (id INTEGER NOT NULL PRIMARY KEY, date TEXT NOT NULL, name TEXT NOT NULL, positionName TEXT NOT NULL, FOREIGN KEY (name) REFERENCES individuals(name), FOREIGN KEY (positionName) REFERENCES position(positionName));
